140pub fn unrolled_find_u16s(needle: u16, haystack: &[u16]) -> Option<usize> {
141    let ptr = haystack.as_ptr();
142    let mut start = haystack;
143
144    // For performance reasons unfold the loop eight times.
145    while start.len() >= 8 {
146        macro_rules! if_return {
147            ($($n:literal,)+) => {
148                $(
149                    if start[$n] == needle {
150                        return Some(((&start[$n] as *const u16).addr() - ptr.addr()) / 2);
151                    }
152                )+
153            }
154        }
155
156        if_return!(0, 1, 2, 3, 4, 5, 6, 7,);
157
158        start = &start[8..];
159    }
160
161    for c in start {
162        if *c == needle {
163            return Some(((c as *const u16).addr() - ptr.addr()) / 2);
164        }
165    }
166    None
167}

169pub fn to_u16s<S: AsRef<OsStr>>(s: S) -> crate::io::Result<Vec<u16>> {
170    fn inner(s: &OsStr) -> crate::io::Result<Vec<u16>> {
171        // Most paths are ASCII, so reserve capacity for as much as there are bytes
172        // in the OsStr plus one for the null-terminating character. We are not
173        // wasting bytes here as paths created by this function are primarily used
174        // in an ephemeral fashion.
175        let mut maybe_result = Vec::with_capacity(s.len() + 1);
176        maybe_result.extend(s.encode_wide());
177
178        if unrolled_find_u16s(0, &maybe_result).is_some() {
179            return Err(crate::io::const_error!(
180                ErrorKind::InvalidInput,
181                "strings passed to WinAPI cannot contain NULs",
182            ));
183        }
184        maybe_result.push(0);
185        Ok(maybe_result)
186    }
187    inner(s.as_ref())
188}

190// Many Windows APIs follow a pattern of where we hand a buffer and then they
191// will report back to us how large the buffer should be or how many bytes
192// currently reside in the buffer. This function is an abstraction over these
193// functions by making them easier to call.
194//
195// The first callback, `f1`, is passed a (pointer, len) pair which can be
196// passed to a syscall. The `ptr` is valid for `len` items (u16 in this case).
197// The closure is expected to:
198// - On success, return the actual length of the written data *without* the null terminator.
199//   This can be 0. In this case the last_error must be left unchanged.
200// - On insufficient buffer space,
201//   - either return the required length *with* the null terminator,
202//   - or set the last-error to ERROR_INSUFFICIENT_BUFFER and return `len`.
203// - On other failure, return 0 and set last_error.
204//
205// This is how most but not all syscalls indicate the required buffer space.
206// Other syscalls may need translation to match this protocol.
207//
208// Once the syscall has completed (errors bail out early) the second closure is
209// passed the data which has been read from the syscall. The return value
210// from this closure is then the return value of the function.
211pub fn fill_utf16_buf<F1, F2, T>(mut f1: F1, f2: F2) -> crate::io::Result<T>
212where
213    F1: FnMut(*mut u16, u32) -> u32,
214    F2: FnOnce(&[u16]) -> T,
215{
216    // Start off with a stack buf but then spill over to the heap if we end up
217    // needing more space.
218    //
219    // This initial size also works around `GetFullPathNameW` returning
220    // incorrect size hints for some short paths:
221    // https://github.com/dylni/normpath/issues/5
222    let mut stack_buf: [MaybeUninit<u16>; 512] = [MaybeUninit::uninit(); 512];
223    let mut heap_buf: Vec<MaybeUninit<u16>> = Vec::new();
224    unsafe {
225        let mut n = stack_buf.len();
226        loop {
227            let buf = if n <= stack_buf.len() {
228                &mut stack_buf[..]
229            } else {
230                let extra = n - heap_buf.len();
231                heap_buf.reserve(extra);
232                // We used `reserve` and not `reserve_exact`, so in theory we
233                // may have gotten more than requested. If so, we'd like to use
234                // it... so long as we won't cause overflow.
235                n = heap_buf.capacity().min(u32::MAX as usize);
236                // Safety: MaybeUninit<u16> does not need initialization
237                heap_buf.set_len(n);
238                &mut heap_buf[..]
239            };
240
241            // This function is typically called on windows API functions which
242            // will return the correct length of the string, but these functions
243            // also return the `0` on error. In some cases, however, the
244            // returned "correct length" may actually be 0!
245            //
246            // To handle this case we call `SetLastError` to reset it to 0 and
247            // then check it again if we get the "0 error value". If the "last
248            // error" is still 0 then we interpret it as a 0 length buffer and
249            // not an actual error.
250            c::SetLastError(0);
251            let k = match f1(buf.as_mut_ptr().cast::<u16>(), n as u32) {
252                0 if api::get_last_error().code == 0 => 0,
253                0 => return Err(crate::io::Error::last_os_error()),
254                n => n,
255            } as usize;
256            if k == n && api::get_last_error().code == c::ERROR_INSUFFICIENT_BUFFER {
257                n = n.saturating_mul(2).min(u32::MAX as usize);
258            } else if k > n {
259                n = k;
260            } else if k == n {
261                // It is impossible to reach this point.
262                // On success, k is the returned string length excluding the null.
263                // On failure, k is the required buffer length including the null.
264                // Therefore k never equals n.
265                unreachable!();
266            } else {
267                // Safety: First `k` values are initialized.
268                let slice: &[u16] = buf[..k].assume_init_ref();
269                return Ok(f2(slice));
270            }
271        }
272    }
273}

279pub fn truncate_utf16_at_nul(v: &[u16]) -> &[u16] {
280    match unrolled_find_u16s(0, v) {

313pub fn dur2timeout(dur: Duration) -> u32 {
314    // Note that a duration is a (u64, u32) (seconds, nanoseconds) pair, and the
315    // timeouts in windows APIs are typically u32 milliseconds. To translate, we
316    // have two pieces to take care of:
317    //
318    // * Nanosecond precision is rounded up
319    // * Greater than u32::MAX milliseconds (50 days) is rounded up to INFINITE
320    //   (never time out).
321    dur.as_secs()
322        .checked_mul(1000)
323        .and_then(|ms| ms.checked_add((dur.subsec_nanos() as u64) / 1_000_000))
324        .and_then(|ms| ms.checked_add(if dur.subsec_nanos() % 1_000_000 > 0 { 1 } else { 0 }))
325        .map(|ms| if ms > <u32>::MAX as u64 { c::INFINITE } else { ms as u32 })
326        .unwrap_or(c::INFINITE)
327}

329/// Use `__fastfail` to abort the process
330///
331/// In Windows 8 and later, this will terminate the process immediately without
332/// running any in-process exception handlers. In earlier versions of Windows,
333/// this sequence of instructions will be treated as an access violation, which
334/// will still terminate the process but might run some exception handlers.
335///
336/// https://docs.microsoft.com/en-us/cpp/intrinsics/fastfail
337#[cfg(not(miri))] // inline assembly does not work in Miri
338pub fn abort_internal() -> ! {
339    unsafe {
340        cfg_if::cfg_if! {
341            if #[cfg(any(target_arch = "x86", target_arch = "x86_64"))] {
342                core::arch::asm!("int $$0x29", in("ecx") c::FAST_FAIL_FATAL_APP_EXIT, options(noreturn, nostack));
343            } else if #[cfg(all(target_arch = "arm", target_feature = "thumb-mode"))] {
344                core::arch::asm!(".inst 0xDEFB", in("r0") c::FAST_FAIL_FATAL_APP_EXIT, options(noreturn, nostack));
345            } else if #[cfg(any(target_arch = "aarch64", target_arch = "arm64ec"))] {
346                core::arch::asm!("brk 0xF003", in("x0") c::FAST_FAIL_FATAL_APP_EXIT, options(noreturn, nostack));
347            } else {
348                core::intrinsics::abort();
349            }
350        }
351    }
352}
